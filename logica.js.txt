document.getElementById('loginForm').addEventListener('submit', function(event) {
    event.preventDefault(); // Impede o envio padrão do formulário

    const email = document.getElementById('email').value;
    const senha = document.getElementById('senha').value;

    // Simulação de envio para o back-end (você precisará de um back-end real para isso)
    fetch('/api/login', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({ email, senha })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            // Autenticação bem-sucedida, redirecionar para a página correta
            if (data.role === 'admin') {
                window.location.href = '/admin/dashboard';
            } else if (data.role === 'employee') {
                window.location.href = '/employee/agendamentos';
            } else {
                window.location.href = '/cliente/agendamentos';
            }
        } else {
            alert('E-mail ou senha incorretos.');
        }
    })
    .catch(error => {
        console.error('Erro:', error);
        alert('Ocorreu um erro ao tentar fazer login.');
    });
});
